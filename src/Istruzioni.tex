\section{Passaggi}
\begin{enumerate}[1.]
    \item Creare un account su \href{https://github.com/signup}{\textbf{Github}};
    \item Aggiungere ai preferiti la \href{https://github.com/Basicnickname/Progetto-Lab-3.git}{repository} e farsi aggiungere come collaboratore;
    \item Creare un nuovo repository (pubblico o privato a seconda delle esigenze);
    \item Clonare il repository in locale (usando VSCode o git bash);
    \item Creare un nuovo file \texttt{.tex} all'interno della cartella clonata;
    \item Scrivere il codice \LaTeX all'interno del file;
    \item Salvare il file e fare il commit delle modifiche;
    \item Pushare le modifiche sul repository remoto.
\end{enumerate}
